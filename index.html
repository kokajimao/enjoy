let holeX = leftWall + (areaWidth + wallThickness)/2 - holeWidth/2; ctx.fillRect(leftWall, bottomWall, holeX-leftWall, wallThickness); ctx.fillRect(holeX+holeWidth, bottomWall, (leftWall+areaWidth+wallThickness)-(holeX+holeWidth), wallThickness); location.reload(); } else { alert("ステージ " + (currentStage + 1)); initStage(); } } } function drawBall(ball) { ctx.beginPath(); ctx.arc(ball.x, ball.y, ballRadius, 0, Math.PI * 2); ctx.fillStyle = "#00ffff"; ctx.fill(); } function drawPaddle() { ctx.fillStyle = "#00ff88"; ctx.fillRect(paddleX, canvas.height - paddleHeight - 10, paddleWidth, paddleHeight); } function drawItems() { items.forEach(item => { ctx.fillStyle = "yellow"; ctx.fillRect(item.x, item.y, 15, 15); }); } // ===== 衝突（完全修正版） ===== function collision(ball) { let holeX = leftWall + (areaWidth + wallThickness)/2 - holeWidth/2; // 画面端 if (ball.x < ballRadius) { ball.x = ballRadius;
